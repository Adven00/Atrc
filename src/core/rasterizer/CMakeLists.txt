CMAKE_MINIMUM_REQUIRED(VERSION 3.10)

PROJECT(RASTERIZER)

FILE(GLOB_RECURSE RASTERIZER_SRC
		"${PROJECT_SOURCE_DIR}/src/*.cpp"
		"${PROJECT_SOURCE_DIR}/src/*.h"
		"${PROJECT_SOURCE_DIR}/include/agz/rasterizer/*.h"
		"${PROJECT_SOURCE_DIR}/include/agz/rasterizer/*.inl")

SET(RASTERIZER_SRC ${RASTERIZER_SRC} PARENT_SCOPE)
SET(Rasterizer_INCLUDE_DIRS "${PROJECT_SOURCE_DIR}/include/" PARENT_SCOPE)
SET(Rasterizer_PROJECT_DIR ${PROJECT_SOURCE_DIR} PARENT_SCOPE)
