CMAKE_MINIMUM_REQUIRED(VERSION 3.10)

PROJECT(AGZ_COMMON)

FILE(GLOB_RECURSE AGZ_COMMON_SRC
		"${PROJECT_SOURCE_DIR}/src/*.cpp"
		"${PROJECT_SOURCE_DIR}/src/*.h"
		"${PROJECT_SOURCE_DIR}/include/agz/common/*.h"
		"${PROJECT_SOURCE_DIR}/include/agz/common/*.inl")

SET(AGZ_COMMON_SRC ${AGZ_COMMON_SRC} PARENT_SCOPE)
SET(AGZCommon_INCLUDE_DIRS "${PROJECT_SOURCE_DIR}/include/" PARENT_SCOPE)
SET(AGZCommon_SOURCE_DIR ${PROJECT_SOURCE_DIR} PARENT_SCOPE)
